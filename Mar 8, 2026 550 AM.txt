require("dotenv").config();
const express = require("express");
const TelegramBot = require("node-telegram-bot-api");
const axios = require("axios");
const Stripe = require("stripe");

const app = express();
app.use(express.json());

// --- CONFIG ---
// Replace these with process.env.VARIABLE_NAME in production
const token = process.env.TELEGRAM_TOKEN || "YOUR_TELEGRAM_TOKEN";
const stripe = new Stripe(process.env.STRIPE_SECRET_KEY || "YOUR_STRIPE_KEY");
const OPENAI_KEY = process.env.OPENAI_API_KEY || "YOUR_OPENAI_KEY";

const bot = new TelegramBot(token, { polling: true });

// TEMP DATABASE
let users = {};

// --- BOT LOGIC ---
bot.on("message", async (msg) => {
  const chatId = msg.chat.id;
  const text = msg.text;

  if (!text) return;

  // Initialize User
  if (!users[chatId]) {
    users[chatId] = { plan: "free", wallet: 0 };
  }
  const user = users[chatId];

  // Commands
  if (text === "/start") {
    return bot.sendMessage(chatId, "🚀 *NVUS STARBURST AI*\n\nMake money with cosmic tools\n\nType /earn to begin\\.", { parse_mode: "MarkdownV2" });
  }

  if (text === "/earn") {
    const menu = "💰 *EARN MENU*\n\n/ai \\- Generate Content\n/leads \\- Find Clients\n/wallet \\- View Balance";
    return bot.sendMessage(chatId, menu, { parse_mode: "MarkdownV2" });
  }

  if (text === "/ai") {
    return bot.sendMessage(chatId, "Send: `ai: your prompt`", { parse_mode: "MarkdownV2" });
  }

  // AI Generation Logic
  if (text.toLowerCase().startsWith("ai:")) {
    if (user.plan !== "premium") {
      return bot.sendMessage(chatId, "✨ *STARBURST PREVIEW*\n\nAI content is ready, but locked\\.\n\n🔓 Unlock full version\n👉 /upgrade", { parse_mode: "MarkdownV2" });
    }

    bot.sendMessage(chatId, "⏳ *Generating cosmic insight...*", { parse_mode: "MarkdownV2" });
    const prompt = text.replace(/ai:/i, "").trim();

    try {
      const res = await axios.post(
        "https://api.openai.com/v1/chat/completions",
        {
          model: "gpt-4o-mini",
          messages: [{ role: "user", content: prompt }]
        },
        { headers: { Authorization: `Bearer ${OPENAI_KEY}` } }
      );
      return bot.sendMessage(chatId, res.data.choices[0].message.content);
    } catch (err) {
      return bot.sendMessage(chatId, "❌ AI error. Check your API Key.");
    }
  }

  if (text === "/leads") {
    return bot.sendMessage(chatId, "Send niche \\+ city\nExample: `plumbers Miami`", { parse_mode: "MarkdownV2" });
  }

  if (text === "/wallet") {
    return bot.sendMessage(chatId, `💰 *Wallet Balance:* $${user.wallet}\n\n_Withdrawals available at $50_`, { parse_mode: "MarkdownV2" });
  }

  if (text === "/upgrade") {
    const paymentUrl = process.env.STRIPE_CHECKOUT_URL || "https://buy.stripe.com/your_link";
    return bot.sendMessage(chatId, `💳 *Upgrade to Premium*\n\nUnlock unlimited AI and Leads access\\.\n\n[Click here to Pay](${paymentUrl})`, { parse_mode: "MarkdownV2" });
  }
});

// --- STRIPE WEBHOOK ---
app.post("/stripe-webhook", (req, res) => {
  const event = req.body;

  if (event.type === "checkout.session.completed") {
    const session = event.data.object;
    // You must pass the chatId in the metadata when creating the Stripe session
    const chatId = session.metadata.chat_id;

    if (chatId && users[chatId]) {
      users[chatId].plan = "premium";
      bot.sendMessage(chatId, "🔥 *PURCHASE SUCCESSFUL!*\n\nYou now have Starburst Elite status.", { parse_mode: "MarkdownV2" });
    }
  }
  res.sendStatus(200);
});

// --- SERVER ---
app.get("/", (req, res) => res.send("Starburst SaaS is Online 🚀"));

const PORT = process.env.PORT || 3000;
app.listen(PORT, () => console.log(`Server running on port ${PORT}`));
